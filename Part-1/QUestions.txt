1. Factorial
#include <iostream>
using namespace std;
class Factorial {
public:
    long calculate(int n) {
        long fact = 1;
        for(int i = 1; i <= n; i++) fact *= i;
        return fact;
    }
};
int main() {
    Factorial obj;
    cout << obj.calculate(5);
    return 0;
}

3. Complex Numbers
#include <iostream>
using namespace std;
class Complex {
    float r, i;
public:
    Complex(float real = 0, float imag = 0) : r(real), i(imag) {}
    void add(Complex c1, Complex c2) { r = c1.r + c2.r; i = c1.i + c2.i; }
    void sub(Complex c1, Complex c2) { r = c1.r - c2.r; i = c1.i - c2.i; }
    void mul(Complex c1, Complex c2) { r = c1.r*c2.r - c1.i*c2.i; i = c1.r*c2.i + c1.i*c2.r; }
    void div(Complex c1, Complex c2) {
        float den = c2.r*c2.r + c2.i*c2.i;
        r = (c1.r*c2.r + c1.i*c2.i)/den;
        i = (c1.i*c2.r - c1.r*c2.i)/den;
    }
    void disp() { cout << r << " + " << i << "i" << endl; }
};
int main() {
    Complex a(4, 5), b(2, 3), res;
    res.add(a, b); res.disp();
    return 0;
}

4. 3x3 Matrix Operations
#include <iostream>
using namespace std;
class Matrix {
    int mat[3][3];
public:
    void accept() { for(int i=0;i<3;i++) for(int j=0;j<3;j++) cin >> mat[i][j]; }
    void add(Matrix m1, Matrix m2) {
        for(int i=0;i<3;i++) {
            for(int j=0;j<3;j++) cout << m1.mat[i][j] + m2.mat[i][j] << " ";
            cout << endl;
        }
    }
    int determinant() {
        int det = 0;
        for(int i=0; i<3; i++)
            det += (mat[0][i] * (mat[1][(i+1)%3] * mat[2][(i+2)%3] - mat[1][(i+2)%3] * mat[2][(i+1)%3]));
        return det;
    }
};

#include <iostream>

    int find(int n) {
        int count = 0;

#include <iostream>

    float a, b, c;

    Quadratic(float x, float y, float z) : a(x), b(y), c(z) {}

13. Distance between Points
#include <iostream>
#include <cmath>
using namespace std;
class Points {
    float x, y;
public:
    Points(float a, float b) : x(a), y(b) {}
    float distance(Points p2) {
        return sqrt((x - p2.x)*(x - p2.x) + (y - p2.y)*(y - p2.y));
    }
};

    char *name; int size;
public:
    String(const char *s) {
        size = strlen(s);
        name = new char[size + 1];
        strcpy(name, s);
